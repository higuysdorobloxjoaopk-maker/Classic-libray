local library = {} library._toggles = {}

local players = game:GetService(“Players”) local player =
players.LocalPlayer

local TurtleUiLib = Instance.new(“ScreenGui”) TurtleUiLib.Name =
“TurtleUiLib” TurtleUiLib.Parent = game.CoreGui

function library:SetToggle(name, state) library._toggles[name] = state
end

function library:GetToggle(name) return library._toggles[name] end

function library:Window(name) local UiWindow = Instance.new(“Frame”)
UiWindow.Parent = TurtleUiLib UiWindow.BackgroundColor3 =
Color3.fromRGB(255,255,255) UiWindow.BackgroundTransparency = 0.75
UiWindow.Size = UDim2.new(0,400,0,300) UiWindow.Position =
UDim2.new(0.5,-200,0.5,-150) UiWindow.Active = true UiWindow.Draggable =
true

    local TopBar = Instance.new("Frame")
    TopBar.Parent = UiWindow
    TopBar.BackgroundColor3 = Color3.fromRGB(255,255,255)
    TopBar.BackgroundTransparency = 0.5
    TopBar.Size = UDim2.new(1,0,0,30)

    local Title = Instance.new("TextLabel")
    Title.Parent = TopBar
    Title.Size = UDim2.new(1,-30,1,0)
    Title.Position = UDim2.new(0,5,0,0)
    Title.BackgroundTransparency = 1
    Title.Text = name or "Window"
    Title.TextColor3 = Color3.fromRGB(0,0,0)
    Title.TextSize = 18
    Title.TextXAlignment = Enum.TextXAlignment.Left

    local Close = Instance.new("TextButton")
    Close.Parent = TopBar
    Close.Size = UDim2.new(0,30,1,0)
    Close.Position = UDim2.new(1,-30,0,0)
    Close.BackgroundTransparency = 1
    Close.Text = "X"
    Close.TextColor3 = Color3.fromRGB(255,0,0)
    Close.TextSize = 18
    Close.MouseButton1Click:Connect(function()
        UiWindow:Destroy()
    end)

    local Container = Instance.new("Frame")
    Container.Parent = UiWindow
    Container.BackgroundTransparency = 1
    Container.Position = UDim2.new(0,0,0,30)
    Container.Size = UDim2.new(1,0,1,-30)

    local layout = Instance.new("UIListLayout")
    layout.Parent = Container
    layout.Padding = UDim.new(0,5)

    local functions = {}

    function functions:Toggle(name, default, callback)
        library._toggles[name] = default

        local ToggleBtn = Instance.new("TextButton")
        ToggleBtn.Parent = Container
        ToggleBtn.Size = UDim2.new(1,-10,0,30)
        ToggleBtn.BackgroundColor3 = Color3.fromRGB(240,240,240)
        ToggleBtn.TextColor3 = Color3.fromRGB(0,0,0)
        ToggleBtn.TextSize = 16

        local function update()
            ToggleBtn.Text = name .. ": " .. (library._toggles[name] and "ON" or "OFF")
        end

        update()

        ToggleBtn.MouseButton1Click:Connect(function()
            library._toggles[name] = not library._togles[name]
            update()
            if callback then
                callback(library._toggles[name])
            end
        end)

        return {
            Set = function(_,state)
                library._toggles[name] = state
                update()
            end,
            Get = function(_)
                return library._toggles[name]
            end
        }
    end

    return functions

end

return library
